# Guianan and Amazonian Spirituality have various different doctrines that are nearly mechanically identical - one doctrine for each religion the doctrine is intended to be on
# This allows the 'normal' hostility level for a given religion to be used for a faith with the doctrine and any custom faith made off of it
# The most pressing example of this is that a custom faith made off of Sanatani Hinduism will see and be seen by Sanatani as Astray (using default same_religion relations) rather than Hostile via an override
# In the Amazon, this also means the Amazonian doctrines don't see one another as Hostile
# The same principle applies to Apostate Afro-Diasporic faiths made off of Winti, and technically applies to Kebatinan and Potay, though their default hostility level is already Hostile
# Please note that the Spiritist version of the Amazonian doctrine does not do any overrides of its own, acting only as a receptor. This is since Spiritists already default to hostile at most for hostilities

# The access doctrines determine whether a faith can see the Amazonian or Guianan set.
doctrine_amazonian_tolerance_access = {
	group = "not_creatable"
	is_available_on_create = {
		OR = {
			has_doctrine = amazonian_access
			has_doctrine = guianan_access
		}
	}

	amazonian_access = {
		parameters = {
			can_select_amazonian_doctrine = yes
		}
	}

	guianan_access = {
		parameters = {
			can_select_guianan_doctrine = yes
		}
	}
}
