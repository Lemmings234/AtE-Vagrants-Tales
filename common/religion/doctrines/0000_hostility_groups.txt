ifs_override_doctrine_group = {
	group = "not_creatable"
	# Like the Syncretism override doctrines below this one, this doctrine exists to make IFS function correctly. 
	# It is written on a separate line so that there's no possible weirdness with having IFS and a syncretism tenet

	is_available_on_create = {
		# These doctrines 'fall off' in faith creation. The faith creation on_action applies the appropriate doctrine if the created faith has Inter-Faith Services
		always = no
	}

	ifs_override_doctrine = {
		parameters = {
			hostility_override_doctrine_pluralism_pluralistic = 0
			hostility_override_doctrine_pluralism_righteous = 1
			hostility_override_doctrine_pluralism_fundamentalist = 1
		}
	}
}
