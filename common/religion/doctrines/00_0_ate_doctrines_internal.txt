doctrine_maya_internal_tolerance = {
	group = "main_group"

	# Maya Internal Tolerance accounts for the unique situation of faiths from multiple religions recognizing the Nohoch Tata, as well as improving relations in the region more generally
	# Maya Harmony is present on: Qabantajik, Votanzapatista, Cruzo'ob
	# Maya Heritage is present on: Tihi'ikoran, Lamanite
	# Maya Modernism is present on: Neomayan
	# Maya Madness is present on: Balam Votan, Hach Kanik (with special localization to "Authentic Maya")

	is_available_on_create = {
		# Only accessible to faiths which already have one of the doctrines. If expanded on, include any added doctrines in this section
		OR = {
			has_doctrine = special_doctrine_maya_harmony
			has_doctrine = special_doctrine_maya_heritage
			has_doctrine = special_doctrine_maya_modernism
			has_doctrine = special_doctrine_maya_madness
		}
	}

	special_doctrine_maya_harmony = {
		# The Nohoch Tata doctrine. Requires Spiritual Head to function properly
		# Sets your HoF to be the Nohoch Tata
		# Sees itself and Heritage as Astray, Modernism as Hostile
		is_shown = {
			always = yes
		}
		can_pick = {
			custom_description = {
				text = doctrine_requires_theocratic_head_of_faith_trigger
				OR = {
					doctrine:doctrine_spiritual_head = { is_in_list = selected_doctrines }
				}
			}
		}
		parameters = {
			follows_nohoch_tata = yes
			hostility_override_special_doctrine_maya_harmony = 1
			hostility_override_special_doctrine_maya_heritage = 1
			hostility_override_special_doctrine_maya_modernism = 2
		}
	}

	special_doctrine_maya_heritage = {
		# the derivative doctrine. Intended for "accepted" faiths in the region that don't fall under the Nohoch Tata's influence
		# Same parameters as Harmony
		parameters = {
			hostility_override_special_doctrine_maya_harmony = 1
			hostility_override_special_doctrine_maya_heritage = 1
			hostility_override_special_doctrine_maya_modernism = 2
		}
	}

	special_doctrine_maya_modernism = {
		# the Mayanist doctrine. Intended for that movement
		# Sees itself as Astray
		# Sees Harmony and Heritage as Hostile
		parameters = {
			hostility_override_special_doctrine_maya_modernism = 1
			hostility_override_special_doctrine_maya_harmony = 2
			hostility_override_special_doctrine_maya_heritage = 2
		}
	}

	special_doctrine_maya_madness = {
		# the "opt-out" doctrine. Has Jungle-specific combat boosts to differentiate it from Apostate
		name = {
			first_valid = {
				triggered_desc = { # for Hach Kanik
					trigger = {
						religion_tag = mesoamerican_religion
					}
					desc = special_doctrine_authentic_maya_name
				}
				desc = special_doctrine_maya_madness_name
			}
		}

		desc = {
			first_valid = {
				triggered_desc = { # for Hach Kanik
					trigger = {
						religion_tag = mesoamerican_religion
					}
					desc = special_doctrine_authentic_maya_desc
				}
				desc = special_doctrine_maya_madness_desc
			}
		}

		character_modifier = {
			jungle_advantage = 5
			jungle_attrition_mult = -0.2
		}
	}
}
