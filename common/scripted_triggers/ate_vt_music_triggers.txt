should_use_british_music_trigger = {
	OR = {
		culture = { has_cultural_pillar = heritage_briton }
		culture = { any_parent_culture_or_above = { has_cultural_pillar = heritage_briton } }
		culture = culture:welsh
		culture = { any_parent_culture_or_above = { this = culture:welsh } }
		religion = religion:britannic_religion
	}
}
should_use_hawaiian_music_trigger = {
	OR = {
		culture = { this = culture:hawaiian }
		culture = { any_parent_culture_or_above = { culture:hawaiian } }
	}
}
