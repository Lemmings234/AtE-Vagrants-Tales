should_use_british_music_trigger = {
	OR = {
		culture = { has_cultural_pillar = heritage_britannic }
		culture = { any_parent_culture_or_above = { has_cultural_pillar = heritage_britannic } }
		culture = { has_cultural_pillar = language_cymraeg }
		culture = { any_parent_culture_or_above = { has_cultural_pillar = language_cymraeg } }
		religion = religion:britannic_religion
	}
}
