tradition_ao_ritual_succession = {
	category = ritual

	layers = {
		0 = learning
		1 = indian
		4 = philosopher.dds
	}

	is_shown = {
		NOT = { has_game_rule = no_adoption }
		OR = {
			has_cultural_tradition = tradition_mystical_ancestors
			any_parent_culture_or_above = {
				OR = {
					has_cultural_tradition = tradition_mystical_ancestors
					has_cultural_tradition = tradition_ao_ritual_succession
				}
			}
		}
	}

	can_pick = { #Can pick if your culture has ritual succession ancestors in its history
		custom_tooltip = {
			text = any_ancestor_culture_was_mystical
			OR = {
				has_cultural_tradition = tradition_mystical_ancestors
				any_parent_culture_or_above = {
					OR = {
						has_cultural_tradition = tradition_mystical_ancestors
						has_cultural_tradition = tradition_ao_ritual_succession
					}
				}
			}
		}
		custom_tooltip = {
			text = cannot_have_tradition_mystical_ancestors
			trigger_if = {
				limit = { NOT = { scope:replacing ?= culture_tradition:tradition_mystical_ancestors } }
				NOT = { culture_tradition:tradition_mystical_ancestors = { is_in_list = traits } }
			}
		}
	}
	
	parameters = {
		ao_lowborn_adoption = yes
		ao_no_prestige_cost_adoption = yes
		ao_legitimate_adoption = yes
		cannot_disherit = yes
		landing_house_members_gives_renown = yes
		piety_penalty_for_revoking_titles_from_house_members = yes
	}

	character_modifier = {
		dynasty_opinion = 10
	}

	cost = {
		prestige = {
			add = {
				value = tradition_double_base_cost
				desc = BASE
			}
			if = {
				limit = {
					NOT = {
						culture_pillar:ethos_spiritual = { is_in_list = traits }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_spiritual_desc
				}
			}
			
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	
	ai_will_do = {
		value = 0
		if = {
			limit = {
				scope:character.culture = {
					any_parent_culture_or_above = {
						has_cultural_tradition = tradition_ao_ritual_succession
					}
				}
			}
			add = 1000
		}
	}
}

tradition_ao_shared_fertility = {
	category = societal

	layers = {
		0 = intrigue
		1 = mena
		4 = greeting.dds
	}
	
	is_shown = {
		NOT = { has_game_rule = no_adoption }
	}
	
	parameters = {
		ao_shared_fertility = yes
		ao_fecund_trait_more_valued = yes
		ao_same_sex_marriage_if_accepted = yes
	}

	character_modifier = {
	}
	
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			if = {
				limit = {
					NOR = {
						culture_pillar:ethos_egalitarian = { is_in_list = traits }
						culture_pillar:ethos_communal = { is_in_list = traits }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_communal_or_egalitarian_desc
				}
			}
			
			multiply = tradition_replacement_cost_if_relevant
		}
	}

	ai_will_do = {
		value = 0
		if = {
			limit = { scope:character = { allowed_to_marry_same_sex_trigger = yes } }
			add = 100
		}
	}
