tradition_RICE_expert_apiculture = {
	category = societal

	layers = {
		0 = steward
		1 = western
		4 = RICE_honey.dds
	}

	is_shown = {
		any_culture_county = {
			count >= 4
			any_county_province = {
				NOT = {
					geographical_region = world_arctic
				}
			}
		}
	}

	can_pick = {
		custom_tooltip = {
			text = culture_not_in_arctic
			any_culture_county = {
				count >= 4
				any_county_province = {
					NOT = {
						geographical_region = world_arctic
					}
				}
			}
		}
	}
	
	parameters = {
		unlock_rice_bee_product_decision = yes
		unlock_rice_bee_tribute_decision = yes
	}

	character_modifier = {
		negate_health_penalty_add = 0.05
		monthly_stewardship_lifestyle_xp_gain_mult = 0.05
	}	

	county_modifier = {
		#development_growth_factor = 0.1
		tax_mult = 0.05
	}
	
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	
	ai_will_do = {
		value = 100
		if = {
			limit = { # AI should avoid stacking traditions for courtly court positions
				culture_tradition:tradition_music_theory = { is_in_list = traits }
			}
			multiply = 0.75
		}
	}
}
