	character_modifier = {

	}

	cost = {
		prestige = {
			add = {

		}

	ai_will_do = {
		value = 25

		if = {
			limit = {

					OR = {
						intrigue > 20
						has_trait = schemer

					}
				}
			}
			add = 100
		}
	}
}
