	trigger = {
		OR = {
			AND = {
				scope:culture = culture:quebecois
				has_primary_title = title:k_nunavut
			}
			AND = {
				scope:culture = culture:ontarien
				has_primary_title = title:k_nunavut
			}
			AND = {
				scope:culture = culture:acadien

			}
			AND = {
				scope:culture = culture:montrealais
				has_primary_title = title:k_nunavut
			}
		}
	}
}
